#include <stdio.h>

#define INF 999999
#define MAX 10

int n;
int cost[MAX][MAX];
int dp[1<<MAX][MAX];   // dp[mask][i] = minimum cost to reach city i using mask

int min(int a, int b) { return a < b ? a : b; }

int tsp(int mask, int pos) {
    if (mask == (1<<n) - 1) 
        return cost[pos][0];  // return to start city

    if (dp[mask][pos] != -1)
        return dp[mask][pos];

    int ans = INF;

    for (int city = 0; city < n; city++) {
        if (!(mask & (1<<city))) {   // city not visited
            int newAns = cost[pos][city] + tsp(mask | (1<<city), city);
            ans = min(ans, newAns);
        }
    }

    return dp[mask][pos] = ans;
}

int main() {
    printf("Enter number of cities: ");
    scanf("%d", &n);

    printf("Enter cost adjacency matrix:\n");
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            scanf("%d", &cost[i][j]);

    // Initialize dp with -1
    for (int i = 0; i < (1<<n); i++)
        for (int j = 0; j < n; j++)
            dp[i][j] = -1;

    int ans = tsp(1, 0);  // start from city 0 with mask = 1 (0001)

    printf("Minimum cost of TSP tour = %d\n", ans);

    return 0;
}
