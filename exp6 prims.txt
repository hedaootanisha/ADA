#include <stdio.h>
#include <limits.h>

#define V 100
#define INF INT_MAX

int main() {
    int n, graph[V][V];
    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix (-1 for no edge):\n");
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++) {
            scanf("%d", &graph[i][j]);
            if (graph[i][j] < 0) graph[i][j] = INF;
        }

    int parent[V], key[V], mst[V] = {0};

    // Initialization
    for (int i = 0; i < n; i++)
        key[i] = INF;

    key[0] = 0;     // Start from node 0
    parent[0] = -1;

    // Primâ€™s Algorithm
    for (int count = 0; count < n - 1; count++) {
        int u = -1, min = INF;

        // Pick minimum key vertex
        for (int i = 0; i < n; i++)
            if (!mst[i] && key[i] < min)
                min = key[i], u = i;

        mst[u] = 1;

        // Update adjacent vertices
        for (int v = 0; v < n; v++)
            if (!mst[v] && graph[u][v] < key[v]) {
                parent[v] = u;
                key[v] = graph[u][v];
            }
    }

    // Output MST
    int total = 0;
    printf("\nEdges in MST:\n");
    for (int i = 1; i < n; i++) {
        printf("%d - %d  (weight %d)\n", parent[i], i, graph[i][parent[i]]);
        total += graph[i][parent[i]];
    }

    printf("Total cost = %d\n", total);
    return 0;
}
